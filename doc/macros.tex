\newcommand\showafter[4]{%
  \ifnum\year>#1\relax
    #4% today.year > args.year
  \else
    \ifnum\year=#1\relax
    % today.year = args.year
      \ifnum\month>#2\relax
      % today.month > args.month
        #4%
      \else
        \ifnum\month=#2\relax
        % today.month = args.month
          \ifnum\day>#3\relax
          % today.day >= args.day
            #4%
          \else
            \ch@removeduplicatespace
          \fi
        \else
          \ch@removeduplicatespace
        \fi
      \fi
    \else
      \ch@removeduplicatespace
    \fi
  \fi}
